57std::optional<signal_t> signal_t::check_signal(const zmq::message_t& msg) noexcept {
58 if (msg.size() != sizeof(uint64_t)) {
59 return std::nullopt;
60 }
61
62 if ((*msg.data<uint64_t>() & ~0xFFULL) != SIGNAL_PREFIX) {
63 return std::nullopt;
64 }
65
66 uint8_t signal_byte = *msg.data<uint64_t>() & 0xFF;
67 switch (signal_byte) {
68 case static_cast<uint8_t>(type_t::success):
69 return signal_t(type_t::success);
70 case static_cast<uint8_t>(type_t::failure):
71 return signal_t(type_t::failure);
72 case static_cast<uint8_t>(type_t::stop):
73 return signal_t(type_t::stop);
74 default:
75 return std::nullopt;
76 }
77}
